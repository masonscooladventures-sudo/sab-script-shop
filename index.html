const express = require("express") const session = require("express-session") const bodyParser = require("body-parser") const app = express() app.use(bodyParser.json()) app.use(session({ secret:"paste-secret", resave:false, saveUninitialized:true })) let users = [] let pastes = [] function detectNSFW(text){ const bad = ["porn","xxx","sex","nsfw"] return bad.some(w=>text.toLowerCase().includes(w)) } const page = ` PasteSite
Home Create
Login Signup
CONTENT
` function render(content){ return page.replace("CONTENT",content) } app.get("/",(req,res)=>{ let list = pastes.map(p=>{ return `

${p.title}

by ${p.user}

` }).join("") let html = `

Pastes

${list} ` res.send(render(html)) }) app.get("/signup",(req,res)=>{ res.send(render(`

Signup

`)) }) app.post("/signup",express.urlencoded({extended:true}),(req,res)=>{ users.push({ user:req.body.user, pass:req.body.pass }) res.redirect("/login") }) app.get("/login",(req,res)=>{ res.send(render(`

Login

`)) }) app.post("/login",express.urlencoded({extended:true}),(req,res)=>{ let u = users.find(x=>x.user===req.body.user && x.pass===req.body.pass) if(!u) return res.send("wrong login") req.session.user = u.user res.redirect("/") }) app.get("/create",(req,res)=>{ if(!req.session.user) return res.redirect("/login") res.send(render(`

Create Paste



`)) }) app.post("/create",express.urlencoded({extended:true}),(req,res)=>{ if(!req.session.user) return res.redirect("/login") let nsfw = req.body.nsfw if(detectNSFW(req.body.text)) nsfw = true let paste = { id:Date.now(), title:req.body.title, text:req.body.text, user:req.session.user, nsfw:nsfw } pastes.unshift(paste) res.redirect("/paste/"+paste.id) }) app.get("/paste/:id",(req,res)=>{ let paste = pastes.find(p=>p.id == req.params.id) if(!paste) return res.send("not found") let content if(paste.nsfw){ content = `

${paste.title}

${paste.text}
` }else{ content = `

${paste.title}

${paste.text}
` } res.send(render(content)) }) app.get("/search",(req,res)=>{ let q = req.query.q.toLowerCase() let results = pastes.filter(p=>p.title.toLowerCase().includes(q) || p.text.toLowerCase().includes(q)) let html = results.map(p=>`

${p.title}

`).join("") res.send(render("

Search results

"+html)) }) app.listen(3000,()=>{ console.log("Server running on http://localhost:3000") })
